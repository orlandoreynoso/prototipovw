SELECT COD_EMPRESA,COD_PROYECTO,COD_BODEGA,COD_PRODUCTO,
CANTIDAD_MINIMA,CANTIDAD_MAXIMA,CANTIDAD_ACTUAL,FECHA_MOVIMIENTO,ESTADO_PRODUCTO
FROM 
DETALLE_BODEGA
WHERE COD_PRODUCTO IN (SELECT COD_PRODUCTO,ID_PRODUCTO,DES_PRODUCTO,PRECIO_PRODUCTO,ESTADO_PRODUCTO FROM
PRODUCTO )



SELECT 
COD_EMPRESA,COD_PROYECTO,COD_BODEGA,CANTIDAD_MINIMA,CANTIDAD_MAXIMA,CANTIDAD_ACTUAL,
FECHA_MOVIMIENTO,PRODUCTO.ESTADO_PRODUCTO,ID_PRODUCTO,DES_PRODUCTO,PRECIO_PRODUCTO
FROM 
DETALLE_BODEGA,PRODUCTO
WHERE DETALLE_BODEGA.COD_PRODUCTO = PRODUCTO.COD_PRODUCTO
